### 시도했지만 실패한 코드
```js
const fs = require('fs')
let input = fs.readFileSync('/dev/stdin').toString().split(' ')

let A = parseInt(input[0])
let B = parseInt(input[1])
let C = parseInt(input[2])

let sale = 0
let i = 0

if (B >= C){
    console.log(-1)
}else {
    while (A + sale >= C) {
        i++
        sale = B*i
    }
    console.log(i)
}
```
